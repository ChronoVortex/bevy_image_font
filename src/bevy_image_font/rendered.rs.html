1//! This module provides functionality for rendering text as images in the Bevy
2//! engine, utilizing custom image fonts. It includes components for
3//! pre-rendering text for both in-world and UI contexts, as well as systems to
4//! update and render the text when changes occur.
5//!
6//! Key Features:
7//! - `ImageFontPreRenderedText` and `ImageFontPreRenderedUiText` components for
8//!   in-world and UI text rendering, respectively.
9//! - Systems for rendering text updates to `Sprite` or `ImageNode` components
10//!   dynamically.
11//! - Integrates with the `image` crate for low-level image manipulation.
12

33/// Internal plugin for conveniently organizing the code related to this
34/// module's feature.
35#[derive(Default)]
36pub(crate) struct RenderedPlugin;

38impl Plugin for RenderedPlugin {
39    fn build(&self, app: &mut App) {
40        app.add_systems(
41            PostUpdate,
42            render_text_to_sprite
43                .after(sync_texts_with_font_changes)
44                .in_set(ImageFontSet),
45        );
46
47        #[cfg(feature = "ui")]
48        {
49            use bevy::ui::widget::update_image_content_size_system;
50            app.add_systems(
51                PostUpdate,
52                render_text_to_image_node
53                    .in_set(ImageFontSet)
54                    .before(update_image_content_size_system)
55                    .after(sync_texts_with_font_changes),
56            );
57        }
58    }
59}

84/// System that renders each [`ImageFontText`] into its [`Sprite`]. This system
85/// only runs when the `ImageFontText` changes.
86pub fn render_text_to_sprite(
87    mut query: Query<(&ImageFontText, &mut Sprite), Changed<ImageFontText>>,
88    image_fonts: Res<Assets<ImageFont>>,
89    mut images: ResMut<Assets<Image>>,
90    layouts: Res<Assets<TextureAtlasLayout>>,
91) {
92    render_text_to_image_holder(
93        query
94            .iter_mut()
95            .map(|(image_font_text, sprite)| (image_font_text, sprite.into_inner())),
96        &image_fonts,
97        &mut images,
98        &layouts,
99    );
100}

121/// Renders text into images and assigns the resulting image handles to their
122/// holders.
123///
124/// This function is designed to work with any type that implements
125/// [`ImageHandleHolder`], allowing it to be used with multiple components, such
126/// as sprites and UI elements.
127///
128/// # Parameters
129/// - `font_text_to_image_iter`: An iterator over pairs of [`ImageFontText`] and
130///   mutable references to objects implementing [`ImageHandleHolder`]. Each
131///   item in the iterator represents a text-to-image mapping to be rendered.
132/// - `image_fonts`: A reference to the font assets used for rendering.
133/// - `images`: A mutable reference to the collection of image assets. This is
134///   used to store the newly rendered images.
135/// - `layouts`: A reference to the collection of texture atlas assets.
136///
137/// The function iterates over the provided items, renders the text for each
138/// [`ImageFontText`], and updates the corresponding [`ImageHandleHolder`] with
139/// the handle to the newly created image.
140///
141/// # Errors
142/// If text rendering fails for an item (e.g., due to missing font assets
143/// or invalid texture layouts), an error message is logged, and the
144/// corresponding holder is not updated.
145fn render_text_to_image_holder<'borrow>(
146    font_text_to_image_iter: impl Iterator<
147        Item = (
148            &'borrow ImageFontText,
149            &'borrow mut (impl ImageHandleHolder + 'borrow),
150        ),
151    >,
152    image_fonts: &Assets<ImageFont>,
153    images: &mut Assets<Image>,
154    layouts: &Assets<TextureAtlasLayout>,
155) {
156    for (image_font_text, image_handle_holder) in font_text_to_image_iter {
157        debug!("Rendering [{}]", image_font_text.text);
158        match render_text_to_image(image_font_text, image_fonts, images, layouts) {
159            Ok(image) => {
160                image_handle_holder.set_image_handle(images.add(image));
161            }
162            Err(error) => {
163                error!(
164                    "Error when rendering image font text {:?}: {}",
165                    image_font_text, error
166                );
167            }
168        }
169    }
170}

172/// Renders the text from an [`ImageFontText`] into a single image.
173///
174/// This function takes a reference to an [`ImageFontText`] component and
175/// generates an image representation of the text. It applies font-specific
176/// filtering, determines appropriate character placements, and composites the
177/// final output.
178///
179/// # Parameters
180/// - `image_font_text`: The text to render, along with its associated font.
181/// - `image_fonts`: The collection of available font assets.
182/// - `images`: The collection of image assets used to retrieve font textures.
183/// - `layouts`: The texture atlas layouts defining character positioning.
184///
185/// # Returns
186/// A [`Result<Image, ImageFontRenderError>`] containing the generated image if
187/// successful, or an error if rendering fails (e.g., due to missing assets).
188///
189/// # Behavior
190/// - If the text is empty, a **1x1 transparent image** is returned to avoid
191///   invalid texture sizes.
192/// - The function leverages [`RenderContext`] to compute character positions
193///   and generate the image.
194#[expect(
195    clippy::cast_possible_truncation,
196    clippy::cast_sign_loss,
197    reason = "numbers are always positive and small enough"
198)]
199fn render_text_to_image(
200    image_font_text: &ImageFontText,
201    image_fonts: &Assets<ImageFont>,
202    images: &Assets<Image>,
203    layouts: &Assets<TextureAtlasLayout>,
204) -> Result<Image, ImageFontRenderError> {
205    let image_font = image_fonts
206        .get(&image_font_text.font)
207        .ok_or(ImageFontRenderError::MissingImageFontAsset)?;
208    let textures = image_font.textures(images);
209
210    let render_config = RenderConfig {
211        text_anchor: Anchor::Center,
212        offset_characters: false,
213        apply_scaling: false,
214        letter_spacing: 0.0,
215        scaling_mode: ImageFontScalingMode::Truncated,
216        color: Color::WHITE, // Currently unused for rendering to an image
217    };
218
219    let mut render_context =
220        RenderContext::new(image_font, image_font_text, render_config, layouts)
221            .ok_or(ImageFontRenderError::MissingTextureAsset)?;
222
223    if render_context.text().is_empty() {
224        // Can't make a 0x0 image, so make a 1x1 transparent black pixel
225        return Ok(Image::new(
226            Extent3d {
227                width: 1,
228                height: 1,
229                depth_or_array_layers: 1,
230            },
231            TextureDimension::D2,
232            vec![0, 0, 0, 0],
233            TextureFormat::Rgba8UnormSrgb,
234            RenderAssetUsages::RENDER_WORLD,
235        ));
236    }
237
238    let width = render_context.text_width() as u32;
239    let height = render_context.max_height();
240
241    let mut output_image = image::RgbaImage::new(width, height);
242    let font_textures: Vec<ImageBuffer<Rgba<u8>, _>> = textures
243        .iter()
244        .map(|texture| {
245            let data = texture.data.as_ref()?;
246            ImageBuffer::from_raw(texture.width(), texture.height(), data.as_slice())
247        })
248        .collect::<Option<_>>()
249        .ok_or(ImageFontRenderError::UnknownError)?;
250
251    let mut x_pos = 0.0;
252    let mut texture_atlas = render_context.font_texture_atlas(' ');
253    let mut color = Color::default();
254    for character in render_context.text().filtered_chars() {
255        let image_font_character = &image_font.atlas_character_map[&character];
256        render_context.update_render_values(character, &mut texture_atlas, &mut color);
257
258        #[expect(
259            clippy::expect_used,
260            reason = "we're using `filtered_chars()` which has only valid characters"
261        )]
262        let rect = texture_atlas
263            .texture_rect(layouts)
264            .expect("`filtered_chars()` guarantees valid characters");
265
266        output_image.copy_from(
267            &*font_textures[image_font_character.page_index].view(
268                rect.min.x,
269                rect.min.y,
270                rect.width(),
271                rect.height(),
272            ),
273            x_pos as u32,
274            0,
275        )?;
276
277        // Let `transform()` handle x-position updates
278        render_context.transform(&mut x_pos, character);
279    }
280
281    #[expect(
282        clippy::cast_possible_truncation,
283        clippy::cast_sign_loss,
284        reason = "the magnitude of the numbers we're working on here are too small to lose anything"
285    )]
286    if let Some(font_height) = image_font_text.font_height {
287        render_context.render_config.apply_scaling = true;
288        let scaled_width = render_context.text_width();
289
290        output_image = imageops::resize(
291            &output_image,
292            scaled_width as u32,
293            font_height as u32,
294            FilterType::Nearest,
295        );
296    }
297
298    let mut bevy_image = Image::new(
299        Extent3d {
300            width: output_image.width(),
301            height: output_image.height(),
302            depth_or_array_layers: 1,
303        },
304        TextureDimension::D2,
305        output_image.into_vec(),
306        TextureFormat::Rgba8UnormSrgb,
307        RenderAssetUsages::RENDER_WORLD,
308    );
309    bevy_image.sampler = ImageSampler::nearest();
310
311    Ok(bevy_image)
312}

314/// Errors that can occur during the rendering of an `ImageFont`.
315#[derive(Debug, Error)]
316#[non_exhaustive]
317pub enum ImageFontRenderError {
318    /// The image could not be converted to a `DynamicImage`. This typically
319    /// indicates an issue with the underlying image data or its format.
320    #[error("failed to convert image to DynamicImage: {0}")]
321    ImageConversion(String),
322
323    /// The `ImageFont` asset required for rendering was not loaded. Ensure
324    /// that the font asset is correctly loaded into the Bevy app. This should
325    /// happend automatically when using the `AssetLoader` to load the font
326    /// asset.
327    #[error("ImageFont asset not loaded")]
328    MissingImageFontAsset,
329
330    /// The texture asset associated with the `ImageFont` was not loaded. This
331    /// could indicate an issue with the asset pipeline or a missing dependency.
332    /// This should happend automatically when using the `AssetLoader` to
333    /// load the font asset.
334    #[error("Font texture asset not loaded")]
335    MissingTextureAsset,
336
337    /// An unspecified internal error occurred during rendering. This may
338    /// indicate a bug or unexpected state in the rendering system.
339    #[error("internal error")]
340    UnknownError,
341
342    /// Failed to copy a character from the source font image texture to the
343    /// target rendered text sprite image texture. This error typically occurs
344    /// when there is an issue with the image data or the copying process.
345    #[error("failed to copy from atlas: {0}")]
346    CopyFailure(#[from] ImageError),
347}

349/// A helper trait that represents types that can hold an image handle.
350///
351/// This is used to abstract over different components (e.g., [`Sprite`] and
352/// [`ImageNode`]) that need to store a handle to an image rendered from text.
353trait ImageHandleHolder {
354    /// Sets the handle for the image that this holder represents.
355    ///
356    /// This method is called after rendering text into an image
357    /// to update the holder with the new image handle.
358    ///
359    /// # Parameters
360    /// - `image`: The handle to the newly rendered image.
361    fn set_image_handle(&mut self, image: Handle<Image>);
362}
363
364impl ImageHandleHolder for Sprite {
365    fn set_image_handle(&mut self, image: Handle<Image>) {
366        self.image = image;
367    }
368}
369
370#[cfg(feature = "ui")]
371impl ImageHandleHolder for ImageNode {
372    fn set_image_handle(&mut self, image: Handle<Image>) {
373        self.image = image;
374    }
